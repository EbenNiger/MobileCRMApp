    <script>
        let contacts = [];
        let currentFilter = 'all';
        let currentSearch = '';
        let editingContactId = null;
        let quickAddMode = false;


        function loadContacts() {
            const stored = localStorage.getItem('crmContacts');
            if (stored) {
                contacts = JSON.parse(stored);
            } else {
                contacts = [
                    {
                        id: Date.now().toString(),
                        name: 'Sample Contact',
                        email: 'sample@example.com',
                        phone: '+1234567890',
                        company: 'Demo Company',
                        source: 'Website',
                        status: 'New Lead',
                        notes: 'This is a sample contact. Tap Edit to modify or Delete to remove.',
                        dateAdded: new Date().toISOString().split('T')[0],
                        history: []
                    }
                ];
                saveContacts();
            }
            renderContacts();
            updateFilterCounts();
        }


        function saveContacts() {
            localStorage.setItem('crmContacts', JSON.stringify(contacts));
        }


        function showSuccessToast(message) {
            const toast = document.getElementById('successToast');
            toast.textContent = message || '✓ Contact saved successfully!';
            toast.classList.add('show');
            setTimeout(() => {
                toast.classList.remove('show');
            }, 2000);
        }


        document.getElementById('quickModeToggle').addEventListener('click', () => {
            quickAddMode = !quickAddMode;
            const btn = document.getElementById('quickModeToggle');
            const overlay = document.getElementById('quickAddOverlay');
            
            if (quickAddMode) {
                btn.classList.add('active');
                overlay.classList.add('active');
                document.getElementById('quickName').focus();
            } else {
                btn.classList.remove('active');
                overlay.classList.remove('active');
            }
        });


        function closeQuickAdd() {
            quickAddMode = false;
            document.getElementById('quickModeToggle').classList.remove('active');
            document.getElementById('quickAddOverlay').classList.remove('active');
            document.getElementById('quickAddForm').reset();
        }


        document.getElementById('quickAddForm').addEventListener('submit', (e) => {
            e.preventDefault();
            
            const newContact = {
                id: Date.now().toString(),
                name: document.getElementById('quickName').value,
                phone: document.getElementById('quickPhone').value,
                email: document.getElementById('quickEmail').value || '',
                company: '',
                source: 'Quick Add',
                status: document.getElementById('quickStatus').value,
                notes: '',
                dateAdded: new Date().toISOString().split('T')[0],
                history: []
            };
            
            contacts.unshift(newContact);
            saveContacts();
            renderContacts();
            updateFilterCounts();
            
            showSuccessToast('⚡ Contact added quickly!');
            document.getElementById('quickAddForm').reset();
            closeQuickAdd();
        });


        function showSetupInstructions() {
            const instructions = document.getElementById('setupInstructions');
            instructions.style.display = instructions.style.display === 'none' ? 'block' : 'none';
        }


        function getStatusClass(status) {
            const classes = {
                'New Lead': 'status-new',
                'Contacted': 'status-contacted',
                'Qualified': 'status-qualified',
                'Negotiating': 'status-negotiating',
                'Converted': 'status-converted',
                'Lost': 'status-lost'
            };
            return classes[status] || 'status-new';
        }


        function renderContacts() {
            const container = document.getElementById('contactsList');
            let filtered = contacts;


            if (currentFilter !== 'all') {
                filtered = filtered.filter(c => c.status === currentFilter);
            }


            if (currentSearch) {
                const search = currentSearch.toLowerCase();
                filtered = filtered.filter(c => 
                    c.name.toLowerCase().includes(search) ||
                    (c.email && c.email.toLowerCase().includes(search)) ||
                    (c.phone && c.phone.toLowerCase().includes(search)) ||
                    (c.company && c.company.toLowerCase().includes(search))
                );
            }


            if (filtered.length === 0) {
                container.innerHTML = `
                    <div class="empty-state">
                        <p style="font-size: 3rem;">👤</p>
                        <p>No contacts found</p>
                        <p style="font-size: 0.875rem;">Try the ⚡ Quick Add button to add contacts faster!</p>
                    </div>
                `;
                return;
            }


            container.innerHTML = filtered.map(contact => `
                <div class="contact-card">
                    <div class="contact-header">
                        <div>
                            <div class="contact-name">${contact.name}</div>
                            ${contact.company ? `
                                <div class="contact-company">
                                    🏢 <span>${contact.company}</span>
                                </div>
                            ` : ''}
                        </div>
                        <span class="status-badge ${getStatusClass(contact.status)}">${contact.status}</span>
                    </div>
                    
                    <div class="contact-info">
                        ${contact.phone ? `
                            <div class="info-row">
                                📞 <a href="tel:${contact.phone}">${contact.phone}</a>
                            </div>
                        ` : ''}
                        ${contact.email ? `
                            <div class="info-row">
                                ✉️ <a href="mailto:${contact.email}">${contact.email}</a>
                            </div>
                        ` : ''}
                        ${contact.source ? `
                            <div class="info-row" style="font-size: 0.75rem; color: #6b7280;">
                                📌 Source: ${contact.source}
                            </div>
                        ` : ''}
                        ${contact.dateAdded ? `
                            <div style="font-size: 0.75rem; color: #9ca3af;">
                                Added: ${contact.dateAdded}
                            </div>
                        ` : ''}
                    </div>
                    
                    ${contact.notes ? `
                        <div class="note-box">
                            <strong>Note:</strong> ${contact.notes}
                        </div>
                    ` : ''}
                    
                    <div class="contact-actions">
                        <button class="action-btn btn-edit" onclick="editContact('${contact.id}')">
                            ✏️ Edit
                        </button>
                        
                        <button class="action-btn btn-delete" onclick="deleteContact('${contact.id}')">
                            🗑️ Delete
                        </button>
                    </div>
                </div>
            `).join('');
        }


        function editContact(id) {
            const contact = contacts.find(c => c.id === id);
            if (!contact) return;
            
            editingContactId = id;

            });

                                📞 <a href="tel:${contact.phone}">${contact.phone}</a>
                            </div>
                        ` : ''}
                        ${contact.email ? `
                            <div class="info-row">
                                ✉️ <a href="mailto:${contact.email}">${contact.email}</a>
                            </div>
                        ` : ''}
                        ${contact.source ? `
                            <div class="info-row" style="font-size: 0.75rem; color: #6b7280;">
                                📌 Source: ${contact.source}
                            </div>
                        ` : ''}
                        ${contact.dateAdded ? `
                            <div style="font-size: 0.75rem; color: #9ca3af;">
                                Added: ${contact.dateAdded}
                            </div>
                        ` : ''}
                    </div>
                    
                    ${contact.notes ? `
                        <div class="note-box">
                            <strong>Note:</strong> ${contact.notes}
                        </div>
                    ` : ''}
                    
                    <div class="contact-actions">
                        <button class="action-btn btn-edit" onclick="editContact('${contact.id}')">
                            ✏️ Edit
                        </button>
                        
                        <button class="action-btn btn-delete" onclick="deleteContact('${contact.id}')">
                            🗑️ Delete
                        </button>
                    </div>
                </div>
            `).join('');
        }


        function editContact(id) {
            const contact = contacts.find(c => c.id === id);
            if (!contact) return;
            
            editingContactId = id;
            document.getElementById('modalTitle').textContent = 'Edit Contact';
            document.getElementById('submitBtnText').textContent = 'Update Contact';
            
            document.getElementById('nameInput').value = contact.name;
            document.getElementById('emailInput').value = contact.email || '';
            document.getElementById('phoneInput').value = contact.phone || '';
            document.getElementById('companyInput').value = contact.company || '';
            document.getElementById('sourceInput').value = contact.source || '';
            document.getElementById('statusInput').value = contact.status;
            document.getElementById('notesInput').value = contact.notes || '';
            
            document.getElementById('contactModal').classList.add('active');
        }


        function deleteContact(id) {
            if (!confirm('Are you sure you want to delete this contact?')) return;
            
            contacts = contacts.filter(c => c.id !== id);
            saveContacts();
            renderContacts();
            updateFilterCounts();
            showSuccessToast('Contact deleted');
        }


        function updateFilterCounts() {
            const buttons = document.querySelectorAll('.filter-btn');
            buttons.forEach(btn => {
                const status = btn.dataset.status;
                let count = 0;
                if (status === 'all') {
                    count = contacts.length;
                } else {
                    count = contacts.filter(c => c.status === status).length;
                }
                btn.textContent = `${status === 'all' ? 'All' : status} (${count})`;
            });
        }
